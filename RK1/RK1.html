def diagnostic_plots(df, variable):
    fig, ax = plt.subplots(figsize=(10,7))
    # 
    plt.subplot(2, 2, 1)
    df[variable].hist(bins=30)
    ## Q-Q plot
    plt.subplot(2, 2, 2)
    stats.probplot(df[variable], dist="norm", plot=plt)
    #   
    plt.subplot(2, 2, 3)
    sns.violinplot(x=df[variable])    
    #   
    plt.subplot(2, 2, 4)
    sns.boxplot(x=df[variable])
    plt.show()
